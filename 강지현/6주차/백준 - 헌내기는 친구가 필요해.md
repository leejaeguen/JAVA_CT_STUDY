import java.util.*;

public class Main {
    static boolean[][] vis;
    static char[][] camp;
    static int n,m;
    static int[] dirx = new int[]{1,0,-1,0};
    static int[] diry = new int[]{0,1,0,-1};
    static int result=0;

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        n = sc.nextInt();
        m = sc.nextInt();
        sc.nextLine();
        camp = new char[607][607];
        vis = new boolean[607][607];
        int x=0;
        int y=0;
        for(int i = 0; i < n; i++){
            String s = sc.nextLine();
            for(int j = 0; j < m; j++){
                camp[i][j] = s.charAt(j);
                if(camp[i][j]=='I'){
                    x=i;y=j;
                }
            }
        }

        bfs(x,y);
        if(result!=0)System.out.println(result);
        else System.out.println("TT");
    }

    private static void bfs(int x,int y) {
        Queue<Node> q = new LinkedList<>();
        q.offer(new Node(x,y));
        vis[x][y] = true;
        while (!q.isEmpty()) {
            Node node = q.poll();
            for (int i = 0; i < 4; i++) {
                int nx = node.x+dirx[i];
                int ny = node.y+diry[i];
                if (nx>=0&&nx<n&&ny>=0&&ny<m&&!vis[nx][ny]&&camp[nx][ny]!='X'){
                vis[nx][ny] = true;
                q.offer(new Node(nx,ny));
                if(camp[nx][ny]=='P')result++;
                }
            }
        }
    }

    static class Node{
        int x;
        int y;
        Node(int x, int y){
            this.x = x;
            this.y = y;
        }
    }
}
