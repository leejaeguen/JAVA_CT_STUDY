import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.*;

public class Main {
    static ArrayList<Node>[] nodeList;     //{노드, 거리}
    static boolean[] vis;

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        nodeList = new ArrayList[10007];
        for (int i = 0; i <= 10001; i++) nodeList[i] = new ArrayList<>();

        ArrayList<Integer> leaf = new ArrayList<>();

        int maximumNum = 0;

        String input;
        while ((input = br.readLine())!= null ) {
            if (!input.isEmpty()) {
                StringTokenizer st = new StringTokenizer(input);

                int a = Integer.parseInt(st.nextToken());
                int b = Integer.parseInt(st.nextToken());
                int c = Integer.parseInt(st.nextToken());

                nodeList[a].add(new Node(b, c));
                nodeList[b].add(new Node(a, c));
            }
        }
        br.close();

        for (int i = 1; i <= 10000; i++) {
            if (nodeList[i].isEmpty()) continue;
            if (nodeList[i].size() == 1) leaf.add(i);
        }

        for (int i = 0; i < leaf.size(); i++) {
            vis = new boolean[10007];

            int resultValue = dfs(leaf.get(i));

            if (maximumNum < resultValue) maximumNum = resultValue;
        }

        System.out.println(maximumNum);
    }

    private static int dfs(int start) {
        vis[start] = true;
        int maxValue = 0;

        for (int i = 0; i < nodeList[start].size(); i++) {
            if (vis[nodeList[start].get(i).n]) continue;
            vis[nodeList[start].get(i).n] = true;

            int getNodeValue = dfs(nodeList[start].get(i).n) + nodeList[start].get(i).e;

            if(maxValue<getNodeValue) maxValue = getNodeValue;
        }
        return maxValue;
    }

    private static class Node {
        int n;
        int e;
        Node(int n, int e) {
            this.n = n;
            this.e = e;
        }
    }
}
