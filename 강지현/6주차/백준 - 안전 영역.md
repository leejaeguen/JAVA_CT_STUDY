import java.util.*;

public class Main {
    static boolean[][] vis;
    static int[][] floor;
    static int n;
    static int[] dirx = new int[]{1, 0, -1, 0};
    static int[] diry = new int[]{0, 1, 0, -1};
    static int result = 0;
    static int maxval = 0;
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        n = sc.nextInt();
        sc.nextLine();
        int result = 1;
        floor = new int[107][107];


        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                floor[i][j] = sc.nextInt();
                if (maxval < floor[i][j]) maxval = floor[i][j];
            }
        }

        for (int f = maxval - 1; f >= 0; f--) {
            vis = new boolean[107][107];
            int cnt=0;
            for (int i = 0; i < n; i++) {
                for (int j = 0; j < n; j++) {
                    if(vis[i][j])continue;
                    if(floor[i][j]<=f)continue;
                    bfs(i,j,f);
                    cnt++;
                }
            }
            if(result<cnt)result=cnt;
        }
        System.out.println(result);
    }
    private static void bfs(int x, int y,int f) {
        Queue<Node> q = new LinkedList<>();
        q.offer(new Node(x, y));
        vis[x][y] = true;
        while (!q.isEmpty()) {
            Node node = q.poll();
            for (int i = 0; i < 4; i++) {
                int nx = node.x + dirx[i];
                int ny = node.y + diry[i];
                if (nx >= 0 && nx < n && ny >= 0 && ny < n && !vis[nx][ny] && floor[nx][ny] > f) {
                    vis[nx][ny] = true;
                    q.offer(new Node(nx, ny));
                }
            }
        }
    }

    static class Node {
        int x;
        int y;

        Node(int x, int y) {
            this.x = x;
            this.y = y;
        }
    }
}
