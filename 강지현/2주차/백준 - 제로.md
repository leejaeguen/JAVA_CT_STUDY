import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        Stack<Integer> stack = new Stack<Integer>();
        for(int i = 0; i < n; i++){
            int temp = sc.nextInt();
            if(temp==0){
                stack.pop();
            }
            else{
                stack.push(temp);
            }
        }
        int total = 0;
        while(stack.size()>0){
            total += stack.pop();
        }
        System.out.println(total);
    }
}
