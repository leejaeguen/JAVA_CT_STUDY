import java.util.Scanner;

public class Main {
    static int[] arr;

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int p = sc.nextInt();
        arr = new int[20];
        for (int k = 1; k <= p; k++) {
            int cnt = 0;
            int t = sc.nextInt();
            for (int i = 0; i < 20; i++) {
                arr[i] = sc.nextInt();
            }
            for (int i = 0; i < 20; i++) {
                for (int j = i + 1; j < 20; j++) {
                    if(arr[i]>arr[j]){
                        cnt++;
                        int temp = arr[i];
                        arr[i] = arr[j];
                        arr[j] = temp;
                    }
                }
            }
            System.out.println(t + " " + cnt);
        }

    }
}
