import java.util.Locale;
import java.util.Scanner;

public class Main {

    static int[] arr;
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        arr = new int[30];
        String s = sc.nextLine();
        s = s.toUpperCase();
        for(int i=0;i<s.length();i++){
            arr[s.charAt(i)-'A']++;
        }
        int maxv=0;
        char maxc = 0;
        boolean flag = true;
        for(int i=0;i<=26;i++){
            if(arr[i]>maxv){
                flag=true;
                maxc = (char)('A'+i);
                maxv=arr[i];
            }
            else if(arr[i]==maxv){
                flag=false;
            }
        }
        if(flag){
            System.out.println(maxc);
        }else{
            System.out.println("?");
        }


    }
}
