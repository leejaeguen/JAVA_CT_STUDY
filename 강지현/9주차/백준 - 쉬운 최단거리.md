import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.LinkedList;
import java.util.Queue;
import java.util.Scanner;
import java.util.StringTokenizer;

public class Main {
    static int[][] arr = new int[1007][1007];
    static boolean[][] vis = new boolean[1007][1007];
    static Queue<Integer> queueX = new LinkedList<>();
    static Queue<Integer> queueY = new LinkedList<>();
    static int[] dx = {1, 0, -1, 0};
    static int[] dy = {0, 1, 0, -1};
    static int n;
    static int m;

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st;

        String nm = br.readLine();
        st = new StringTokenizer(nm, " ");
        n = Integer.parseInt(st.nextToken());
        m = Integer.parseInt(st.nextToken());
        int stx = 0;
        int sty = 0;
        for (int i = 0; i < n; i++) {
            String ln = br.readLine();
            st = new StringTokenizer(ln, " ");
            int j = 0;
            while (st.hasMoreTokens()) {
                arr[i][j] = Integer.parseInt(st.nextToken());
                if (arr[i][j] == 2) {
                    stx = i;
                    sty = j;
                    arr[i][j] = 0;
                }
                j++;
            }
        }


        bfs(stx, sty);

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (!vis[i][j] && arr[i][j] != 0) {
                    System.out.print(-1 + " ");
                } else System.out.print(arr[i][j] + " ");
            }
            System.out.println();
        }

    }

    private static void bfs(int stx, int sty) {
        queueX.offer(stx);
        queueY.offer(sty);
        vis[stx][sty] = true;
        int dis = 1;
        while (!queueX.isEmpty()) {
            int size = queueX.size();
            while (size-- > 0) {
                int ax = queueX.poll();
                int ay = queueY.poll();
                for (int i = 0; i < 4; i++) {
                    int x = ax + dx[i];
                    int y = ay + dy[i];
                    if (x >= 0 && x < n &&
                            y >= 0 && y < m &&
                            !vis[x][y] &&
                            arr[x][y] == 1) {
                        vis[x][y] = true;
                        arr[x][y] = dis;
                        queueX.offer(x);
                        queueY.offer(y);
                    }
                }
            }
            dis++;
        }
    }


}
