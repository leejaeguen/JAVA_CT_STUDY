import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        n = 1000 - n;
        int total=0;
        int[] arr = new int[]{500,100,50,10,5,1};
        for(int i=0;i<6;i++){
            if(n>=arr[i]){
                total+=n/arr[i];
                n=n%arr[i];
            }
        }
        System.out.println(total);
    }
}
