import java.util.*;

public class Main {
    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        List<Integer> list = new ArrayList<>();
        int total = 0;
        for(int i=0; i<n; i++) {
            int a = sc.nextInt();
            total += a;
            list.add(a);
        }
        Collections.sort(list);
        long result = 0;
        int idx=0;
        for(int i=0;i<list.size();i++) {
            total -= list.get(i);
            result += total * list.get(i);
        }
        System.out.println(result);
    }
}
