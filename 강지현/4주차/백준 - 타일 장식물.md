import java.util.*;

public class Main {
    static int n;
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        n = sc.nextInt();
        long[] arr = new long[10007];
        arr[0] = 0;
        arr[1] = 1;
        for(int i=2;i<=n+1;i++){
            arr[i] = arr[i-1] + arr[i-2];
        }
        System.out.println(arr[n+1]*2+arr[n]*2);
    }
}

// ※BigInteger로 구현하기
import java.math.BigInteger;
import java.util.*;

public class Main {
    static int n;

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        n = sc.nextInt();
        BigInteger[] bigArr = new BigInteger[100007];
        bigArr[0] = BigInteger.ZERO;
        bigArr[1] = BigInteger.ONE;
        for (int i = 2; i <= n; i++) {
            bigArr[i] = bigArr[i - 1].add(bigArr[i - 2]);
        }
        System.out.println(bigArr[n]);
    }
}
